  <script>
    var grade = prompt("Enter the Grade");
    var result = "";

    if (grade === null || grade.trim() === "") {
      result = "No grade entered.";
    } else {
      grade = parseFloat(grade);
      if (isNaN(grade) || grade > 100 || grade < 0) {
        result = "INVALID";
      } else if (grade >= 98) {
        result = "WITH HIGHEST HONORS";
      } else if (grade >= 95) {
        result = "WITH HIGH HONORS";
      } else if (grade >= 90) {
        result = "WITH HONORS";
      } else if (grade >= 85) {
        result = "SATISFACTORY";
      } else if (grade >= 80) {
        result = "VERY GOOD";
      } else if (grade >= 75) {
        result = "GOOD";
      } else {
        result = "FAILED";
      }
    }

    document.getElementById("result").textContent = result;
  </script>
